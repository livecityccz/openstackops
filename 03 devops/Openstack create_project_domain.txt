#!/bin/sh


user_name="user_name"

echo ""
printf "Please input \33[;31mUserName:\33[0m(default is "${user_name}"):"
read -t 60 user_name_input
if [ "$user_name_input" != "" ]
then
user_name=${user_name_input}
fi

domain_name="${user_name}"
tenant_name="${user_name}"
tenant_description="${user_name}'s description"
password="${user_name}@2016CloudMind"

email="${user_name}@test.com"
echo ""
printf "Please input \33[;31mEmail:\33[0m(default is "${email}"):"
read -t 60 email_input
if [ "$email_input" != "" ]
then
email=${email_input}
fi

mobile="13111111111"
echo ""
printf "Please input \33[;31mMobile No.:\33[0m(default is "${mobile}"):"
read -t 60 mobile_input
if [ "$mobile_input" != "" ]
then
mobile=${mobile_input}
fi

mysqlcli="mysql -e"
echo ""
printf "Please input \33[;31mMySQL root user's password,and press [Enter] for no:\33[0m(default is "${mysql_root_pass}"):"
read -t 60 mysql_root_pass_input
if [ "${mysql_root_pass_input}" != "" ]
then
mysql_root_pass=${mysql_root_pass_input}
mysqlcli="mysql -uroot -p${mysql_root_pass} -e"
fi


${mysqlcli} \
"use keystone;insert into domain(id,name,enabled,extra) \
values('"${domain_name}"','"${domain_name}"','"1"','{\"description\": \""${domain_name}"\'s description\"}');"
if [ $? == 0 ]
then
  echo "Create ${domain_name} Successful!";
else
  echo "Create ${domain_name} Failed!";
fi


source /root/openrc

keystone tenant-create --name ${tenant_name} --description "${tenant_description}" --enabled true
keystone user-create --name ${user_name} --tenant ${tenant_name} --pass ${password} --email ${email} --tel ${mobile}

${mysqlcli} \
"use keystone;update project set domain_id = '"${domain_name}"' where name = '"${tenant_name}"';"

if [ $? == 0 ]
then
  echo "Add ${tenant_name} to ${domain_name} Successful!";
else
  echo "Add ${tenant_name} to ${domain_name} Failed!";
fi

${mysqlcli} \
"use keystone;update user set domain_id = '"${domain_name}"' where name = '"${user_name}"';"
if [ $? == 0 ]
then
  echo "Add ${user_name} to ${domain_name} Successful!";
else
  echo "Add ${user_name} to ${domain_name} Failed!";
fi


user_rc_dir="/root/o/user_rc/"

echo ""
printf "Please input \33[;31mUser's rc file directory:\33[0m(default is "${user_rc_dir}"):"
read user_rc_dir_input
if [ "$user_rc_dir_input" != "" ]
then
user_rc_dir=$user_rc_dir_input
fi

if [ ! -d ${user_rc_dir} ];
  then
    echo "Destination Directory not exist, Make it";
    mkdir -p ${user_rc_dir};
else
  echo "Destination Directory is OK";
fi

user_rc_filename="${user_name}rc"
user_rc_file="${user_rc_dir}${user_rc_filename}"

(
cat <<eof
#!/bin/sh
export LC_ALL=C
export OS_NO_CACHE='true'
export OS_PROJECT_DOMAIN_NAME="${domain_name}"
export OS_TENANT_DOMAIN_NAME="${domain_name}"
export OS_USER_DOMAIN_NAME="${domain_name}"
export OS_TENANT_NAME="${tenant_name}"
export OS_USERNAME="${user_name}"
export OS_PASSWORD="${password}"
export OS_AUTH_URL='http://192.168.10.2:5000/v2.0/'
export OS_AUTH_STRATEGY='keystone'
export OS_REGION_NAME='RegionOne'
export CINDER_ENDPOINT_TYPE='publicURL'
export GLANCE_ENDPOINT_TYPE='publicURL'
export KEYSTONE_ENDPOINT_TYPE='publicURL'
export NOVA_ENDPOINT_TYPE='publicURL'
export NEUTRON_ENDPOINT_TYPE='publicURL'
eof
) >> ${user_rc_file}


#source /root/openrc
#tenant_id=`keystone tenant-list|grep ${tenant_name}|cut -d ' ' -f 2`
tenant_id=`${mysqlcli} "use keystone;select * from project;"|grep ${tenant_name}|cut -f 1`

neutron net-create --tenant-id ${tenant_id} switch-${tenant_name}
neutron subnet-create  --tenant-id ${tenant_id} switch-${tenant_name} --name subnet-${tenant_name} \
--gateway 10.1.1.1 10.1.1.0/24 \
--dns-nameserver 8.8.4.4 \
--dns-nameserver 1.2.4.8 \
--dns-nameserver 114.114.114.114
neutron router-create --tenant-id ${tenant_id} router-${tenant_name}
neutron router-interface-add router-${tenant_name} subnet-${tenant_name}
neutron router-gateway-set router-${tenant_name} net04_ext
